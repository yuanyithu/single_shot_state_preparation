共享数据结构 linear_section_data

# linear_section_data 是一个 dict，完整描述论文 Eq.(144) 的 E 和 Π:
#
#   "row_xor_steps":      list，表示 E 的行操作序列，按顺序应用到 I_m 即得到 E
#                          每个元素是以下两种三元组之一：
#                            ("swap_rows", row_index_a, row_index_b)
#                            ("xor_rows",  target_row_index, source_row_index)
#                          约定 target_row ^= source_row （逐位 XOR）
#
#   "column_permutation": np.ndarray[int32] shape (num_qubits,)
#                          表示 Π。约定：排列后的第 k 列 = H_Z 的第 column_permutation[k] 列
#                          等价于 Π 矩阵在位置 (column_permutation[k], k) 取 1
#
#   "rank":               int，即论文中的 ρ = rank(H_Z)
#   "num_checks":         int，m
#   "num_qubits":         int，n

import numpy as np


def build_linear_section(parity_check_matrix):
    """
    对 H_Z 做 GF(2) 高斯消元 + 列置换，得到 E · H_Z · Π = [[I_ρ, A], [0, 0]]
    以 XOR 脚本形式记录 E，以排列数组形式记录 Π。
    对应论文 Eq.(144)~(147)。

    输入:
      parity_check_matrix: np.ndarray shape (num_checks, num_qubits), dtype=bool
    返回:
      linear_section_data: dict (格式见上)

    说明: 该 dense 实现对中等尺寸足够用。若 H_Z 很稀疏 (e.g. LDPC, n > 10^4)，
          应换成 scipy.sparse + 最小度选主元；脚本格式不变。
    """
    num_checks, num_qubits = parity_check_matrix.shape

    # 工作矩阵（会被就地消元）
    working_matrix = parity_check_matrix.copy().astype(bool)

    # 列置换初始化为恒等
    column_permutation = np.arange(num_qubits, dtype=np.int32)

    row_xor_steps = []

    # 依次确定第 0, 1, 2, ... 个 pivot
    current_pivot = 0
    while current_pivot < min(num_checks, num_qubits):

        # 在子矩阵 working_matrix[current_pivot:, current_pivot:] 里找一个 1
        # 简单策略: 先扫行，再在该行找列。稀疏版可换成最小度选主元。
        pivot_row_index = -1
        pivot_col_index = -1
        for row_index in range(current_pivot, num_checks):
            nonzero_columns = np.flatnonzero(working_matrix[row_index, current_pivot:])
            if nonzero_columns.size > 0:
                pivot_row_index = row_index
                pivot_col_index = current_pivot + int(nonzero_columns[0])
                break

        if pivot_row_index == -1:
            # 子矩阵全零，rank 已经确定
            break

        # 1) 把 pivot 行换到 current_pivot 位置
        if pivot_row_index != current_pivot:
            working_matrix[[current_pivot, pivot_row_index]] = \
                working_matrix[[pivot_row_index, current_pivot]]
            row_xor_steps.append(("swap_rows", current_pivot, pivot_row_index))

        # 2) 把 pivot 列换到 current_pivot 位置 (只动 working_matrix 和排列，不进 row_xor_steps)
        if pivot_col_index != current_pivot:
            working_matrix[:, [current_pivot, pivot_col_index]] = \
                working_matrix[:, [pivot_col_index, current_pivot]]
            column_permutation[[current_pivot, pivot_col_index]] = \
                column_permutation[[pivot_col_index, current_pivot]]

        # 3) 消去其他所有行里 pivot 列上的 1 (产生 reduced row echelon form)
        for other_row in range(num_checks):
            if other_row == current_pivot:
                continue
            if working_matrix[other_row, current_pivot]:
                working_matrix[other_row] ^= working_matrix[current_pivot]
                row_xor_steps.append(("xor_rows", other_row, current_pivot))

        current_pivot += 1

    rank = current_pivot

    linear_section_data = {
        "row_xor_steps":      row_xor_steps,
        "column_permutation": column_permutation,
        "rank":               rank,
        "num_checks":         num_checks,
        "num_qubits":         num_qubits,
    }
    return linear_section_data


def apply_linear_section(syndrome_bits, linear_section_data):
    """
    计算 r(σ) = Π · [σ_hat; 0]，其中 σ_hat 是 (E·σ) 的前 ρ 位。

    输入:
      syndrome_bits:       np.ndarray shape (num_checks,) dtype=bool，必须 ∈ im H_Z
      linear_section_data: dict
    返回:
      chain_bits:          np.ndarray shape (num_qubits,) dtype=bool，满足 H_Z · chain_bits = σ
    """
    rank               = linear_section_data["rank"]
    num_qubits         = linear_section_data["num_qubits"]
    row_xor_steps      = linear_section_data["row_xor_steps"]
    column_permutation = linear_section_data["column_permutation"]

    # 第一步: transformed = E · syndrome_bits  (按顺序跑脚本)
    transformed = syndrome_bits.copy().astype(bool)
    for op in row_xor_steps:
        if op[0] == "swap_rows":
            _, index_a, index_b = op
            temp = transformed[index_a]
            transformed[index_a] = transformed[index_b]
            transformed[index_b] = temp
        else:  # "xor_rows"
            _, target_row, source_row = op
            transformed[target_row] ^= transformed[source_row]

    # 第二步: 前 rank 位保留，后 (num_checks - rank) 位丢弃; zero-pad 到长度 num_qubits
    # (若 syndrome_bits ∈ im H_Z，后面那些位本就是 0；否则此处会默默裁掉 meta-check violation)
    padded_short = np.zeros(num_qubits, dtype=bool)
    padded_short[:rank] = transformed[:rank]

    # 第三步: chain_bits = Π · padded_short
    # 约定: chain_bits[column_permutation[k]] = padded_short[k]
    chain_bits = np.zeros(num_qubits, dtype=bool)
    chain_bits[column_permutation] = padded_short

    return chain_bits


def apply_linear_section_transpose(chain_like_vector, linear_section_data):
    """
    计算 r^T · v ∈ F_2^m。
    这是 build_logical_observable_masks 里构造 Λ_u = (I + r·H_Z)^T · z_bar_u 的唯一非平凡步骤:
         Λ_u = z_bar_u XOR (H_Z^T · r^T · z_bar_u)
    其中 r^T · z_bar_u 就是本函数的输出。

    输入:
      chain_like_vector:   np.ndarray shape (num_qubits,) dtype=bool  (论文中的 z_bar_u)
      linear_section_data: dict
    返回:
      syndrome_like_vector: np.ndarray shape (num_checks,) dtype=bool  (r^T · z_bar_u)
    """
    rank               = linear_section_data["rank"]
    num_checks         = linear_section_data["num_checks"]
    row_xor_steps      = linear_section_data["row_xor_steps"]
    column_permutation = linear_section_data["column_permutation"]

    # 第一步: intermediate_n = Π^T · chain_like_vector  ∈ F_2^n
    # 推导: 由正向约定 (Π y)[column_permutation[k]] = y[k]，
    #       故 (Π^T v)[k] = v[column_permutation[k]]。
    intermediate_n = chain_like_vector[column_permutation]

    # 第二步: intermediate_m = P^T · intermediate_n  ∈ F_2^m
    # P 是前 ρ 行 = I_ρ 的嵌入矩阵；P^T 取前 ρ 位，再 zero-pad 到长度 m
    intermediate_m = np.zeros(num_checks, dtype=bool)
    intermediate_m[:rank] = intermediate_n[:rank]

    # 第三步: syndrome_like_vector = E^T · intermediate_m
    # E = M_k · M_{k-1} · ... · M_1 (row_xor_steps 记录的顺序即是 M_1, M_2, ..., M_k)
    # 所以 E^T = M_1^T · M_2^T · ... · M_k^T，要倒序应用脚本
    # 对每一步的转置规则:
    #   ("swap_rows", a, b)       -> 自己对称，转置仍是 swap(a, b)
    #   ("xor_rows", target, src) -> 原本是 y[target] ^= y[src]
    #                                转置是 y[src] ^= y[target]  (target/src 角色互换)
    syndrome_like_vector = intermediate_m
    for op in reversed(row_xor_steps):
        if op[0] == "swap_rows":
            _, index_a, index_b = op
            temp = syndrome_like_vector[index_a]
            syndrome_like_vector[index_a] = syndrome_like_vector[index_b]
            syndrome_like_vector[index_b] = temp
        else:  # "xor_rows"
            _, target_row, source_row = op
            # 关键: 这里是 source 收 target，不是正向的 target 收 source
            syndrome_like_vector[source_row] ^= syndrome_like_vector[target_row]

    return syndrome_like_vector

def verify_linear_section(parity_check_matrix, linear_section_data,
                          num_random_tests=20, rng=None):
    """
    随机生成 σ ∈ im H_Z 和任意 v ∈ F_2^n，检查:
      (A) H_Z · r(σ) == σ                                (r 确实是 section)
      (B) <v, r(σ)> == <r^T v, σ>                         (转置关系正确)
    """
    if rng is None:
        rng = np.random.default_rng(0)

    parity_int = parity_check_matrix.astype(np.uint8)
    num_checks, num_qubits = parity_check_matrix.shape

    for _ in range(num_random_tests):
        # 造一个一定属于 im H_Z 的 σ
        random_chain = rng.integers(0, 2, size=num_qubits).astype(bool)
        syndrome_in_image = (parity_int @ random_chain.astype(np.uint8)) % 2
        syndrome_in_image = syndrome_in_image.astype(bool)

        recovered_chain = apply_linear_section(syndrome_in_image, linear_section_data)
        recovered_syndrome = (parity_int @ recovered_chain.astype(np.uint8)) % 2
        assert np.array_equal(syndrome_in_image, recovered_syndrome.astype(bool)), \
            "section 破坏: H_Z r(σ) ≠ σ"

        # 转置配对: 任取 v ∈ F_2^n
        random_v = rng.integers(0, 2, size=num_qubits).astype(bool)
        left_side  = bool(np.bitwise_xor.reduce(random_v & recovered_chain))  # <v, r(σ)>
        rt_v = apply_linear_section_transpose(random_v, linear_section_data)
        right_side = bool(np.bitwise_xor.reduce(rt_v & syndrome_in_image))    # <r^T v, σ>
        assert left_side == right_side, "转置关系破坏: <v, r(σ)> ≠ <r^T v, σ>"
